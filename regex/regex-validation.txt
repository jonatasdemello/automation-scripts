	^([0-9A-Fa-f]{2}[:-]){5}([0-9A-Fa-f]{2})$


CEP:			/^[0-9]{5}-[0-9]{3}$/
CPF:			/^([0-9]{3}\.){2}[0-9]{3}-[0-9]{2}$/
Datas YYYYMMDD: /^20[0-9][0-9]([012][0-9]|[3][01])([0][0-9]|[1][012])$/

IP: 			/^(([1]?[0-9]{1,2}|2([0-4][0-9]|5[0-5]))\.){3}([1]?[0-9]{1,2}|2([0-4][0-9]|5[0-5]))$/
Email: 			/^([[:alnum:]]{1,}|[[:punct:]]{1,})@[[:lower:]]+\.[[:lower:]]+$/


Date
